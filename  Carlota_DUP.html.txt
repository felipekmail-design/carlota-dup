// Session history per workout: key = 'sess_PUSH-A', value = array of sessions (max 4)
function getSessions(wkId){try{return JSON.parse(localStorage.getItem('sess_'+wkId)||'[]');}catch{return[];}}
function saveSessions(wkId,sessions){localStorage.setItem('sess_'+wkId,JSON.stringify(sessions));}

function saveSession(wkId, sessionData){
  const sessions=getSessions(wkId);
  sessions.unshift(sessionData); // newest first
  if(sessions.length>4) sessions.pop(); // keep last 4
  saveSessions(wkId,sessions);
}

// ── HISTORY PANEL HTML ────────────────────────────────────────────────────────
function buildHistPanel(wkId, exList){
  const sessions=getSessions(wkId);
  if(!sessions.length){
    return `<div class="hist-panel">
      <button class="hist-panel-hdr" type="button" onclick="toggleHistPanel(this)">
        <span class="hist-label">📋 Últimas Sessões</span>
        <span class="hist-arrow">▼</span>
      </button>
      <div class="hist-body">
        <div class="no-hist">NENHUMA SESSÃO REGISTRADA AINDA</div>
      </div>
    </div>`;
  }

  const sessHtml = sessions.map((sess,si)=>{
    const exRows = sess.exercises.map((ex,ei)=>{
      if(!ex.sets.some(s=>s.kg||s.reps)) return '';
      const setRows = ex.sets.map((s,i)=>`
        <div class="hist-row-data">
          <div class="hist-sn">${i+1}</div>
          <div class="hist-val${s.kg?'':' empty'}">${s.kg||'—'}</div>
          <div class="hist-val${s.reps?'':' empty'}">${s.reps||'—'}</div>
        </div>`).join('');
      return `<div class="hist-ex">
        <div class="hist-ex-name">${exList[ei]?.name||''}</div>
        <div class="hist-sets">
          <div class="hist-col">Sér</div>
          <div class="hist-col">KG</div>
          <div class="hist-col">Reps</div>
        </div>
        ${setRows}
        ${ex.obs?`<div class="hist-obs">${ex.obs}</div>`:''}
      </div>`;
    }).filter(Boolean).join('');

    return `<div style="margin-bottom:${si<sessions.length-1?'18px':'0'}">
      <div style="font-family:'DM Mono',monospace;font-size:9px;letter-spacing:2px;color:#10b981;text-transform:uppercase;margin-bottom:10px;font-weight:500;">
        ${si===0?'🟢 ÚLTIMA SESSÃO':'SESSÃO -'+(si+1)} · ${sess.date}
      </div>
      ${exRows||'<div class="no-hist">Sem cargas registradas</div>'}
    </div>`;
  }).join('<hr style="border:none;border-top:1px solid #f0f0e8;margin:12px 0;">');

  return `<div class="hist-panel">
    <button class="hist-panel-hdr" type="button" onclick="toggleHistPanel(this)">
      <div>
        <span class="hist-label">📋 Últimas Sessões</span>
        <div class="hist-date" style="margin-top:2px">${sessions[0].date}</div>
      </div>
      <span class="hist-arrow">▼</span>
    </button>
    <div class="hist-body">
      ${sessHtml}
    </div>
  </div>`;
}

function toggleHistPanel(btn){
  const body=btn.nextElementSibling;
  const arr=btn.querySelector('.hist-arrow');
  const open=body.classList.toggle('open');
  arr.textContent=open?'▲':'▼';
}

function openSendModal(wi){
  const w=WK[wi];
  const pg=document.querySelectorAll('.page')[wi];
  const date=new Date().toLocaleDateString('pt-BR',{weekday:'long',day:'2-digit',month:'2-digit',year:'numeric'});
  const subject=`[CARLOTA DUP] ${w.title} — ${date}`;
  let lines=[subject,`Zona: ${w.zoneLabel} | ${w.series}x${w.reps} | Descanso: ${w.rest}`,``];
  let hasData=false;

  // Collect session data for local save
  const sessionData={date:new Date().toLocaleDateString('pt-BR'),exercises:[]};

  pg.querySelectorAll('.ex-card').forEach((card,ci)=>{
    const ex=w.ex[ci];
    const exSession={name:ex.name,sets:[],obs:''};
    let sl=[];

    card.querySelectorAll('.set-row').forEach((row,si)=>{
      const kg=row.querySelectorAll('.inp')[0]?.value||'';
      const rp=row.querySelectorAll('.inp')[1]?.value||'';
      const ok=row.querySelector('.ok-btn')?.classList.contains('done')?'✓':'·';
      exSession.sets.push({kg,reps:rp,done:ok==='✓'});
      if(kg||rp){sl.push(`  S${si+1}: ${kg||'?'}kg × ${rp||'?'} reps ${ok}`);hasData=true;}
    });

    const obs=card.querySelector('.obs')?.value||'';
    exSession.obs=obs;
    sessionData.exercises.push(exSession);

    if(sl.length||obs){
      lines.push(`${ex.n}. ${ex.name}`);
      lines.push(...sl);
      if(obs)lines.push(`  Obs: ${obs}`);
      lines.push('');
    }
  });

  if(!hasData){alert('Registre pelo menos uma carga antes de enviar.');return;}

  emailContent=lines.join('\n');
  pendingSave={wkId:w.id,sessionData,wi};

  document.getElementById('modal-preview').textContent=emailContent;
  document.getElementById('copied-msg').style.display='none';

  const to='felipekmail%40gmail.com';
  const su=encodeURIComponent(subject);
  const body=encodeURIComponent(emailContent);
  document.getElementById('gmail-link').href=
    `https://mail.google.com/mail/?view=cm&fs=1&to=${to}&su=${su}&body=${body}`;

  // Save session immediately when modal opens
  saveSession(pendingSave.wkId, pendingSave.sessionData);
  refreshHistPanel(pendingSave.wi);

  document.getElementById('overlay').classList.add('open');
}

async function copyContent(){
  const msg=document.getElementById('copied-msg');
  try{await navigator.clipboard.writeText(emailContent);}
  catch{
    const ta=document.createElement('textarea');
    ta.value=emailContent;ta.style.cssText='position:fixed;opacity:0;top:0;left:0;';
    document.body.appendChild(ta);ta.focus();ta.select();
    try{document.execCommand('copy');}catch{}
    document.body.removeChild(ta);
  }
  msg.style.display='block';setTimeout(()=>{msg.style.display='none';},3000);
}
